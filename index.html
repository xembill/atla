    Kullanım: ";
        atla                 : Dizini Yazar";
        atla [index]         : Dizini Index Ayarlar";
        atla add             : Bulunduğun Dizini Ekler";
        atla add [dir]       : Dizin Ekler";
        atla rm              : Bulunduğun Dizini Siler";
        atla rm [index]      : Index Dizinini Siler";
        atla save <filename> : Ayarları Dosyaya Yazar";
        atla load <filename> : Ayarları Dosyadan Alır";
        atla clean           : Herşeyi Siler";
        atla help            : Kullanım Klavuzu";

    Key Map (normal):";
        CTRL + Yukarı OK       : Öncekine Atlar";
        CTRL + Aşağı OK     : Sonrakine Atlar";
        CTRL + Sol OK     : Aşağı Atlar";
        CTRL + Sağ OK    : Yukarı Atlar";

    Key Map (putty + screen):";
        ALT + Yukarı OK       :  Öncekine Atlar";
        ALT + Aşağı OK     : Sonrakine Atlar";
        ALT + Sol OK     : Aşağı Atlar";
        ALT + Sağ OK    : Yukarı Atlar";

    Key Map (mac os):";
        ESC + Yukarı OK       :Öncekine Atlar";
        ESC + Aşağı OK     : Sonrakine Atlar";
        ESC + Sol OK     : Aşağı Atlar";
        ESC + Sağ OK    : Yukarı Atlar";
